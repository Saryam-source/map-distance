async function calculateAndSaveTrip(){ var name = document.getElementById('person-name').value.trim(); var from = document.getElementById('from-location').value.trim(); var to = document.getElementById('to-location').value.trim(); if(!name || !from || !to){ toast('Please fill all fields'); return; } setStatus( 'tracking', 'Fetching GPS...', 'Please wait' ); navigator.geolocation.getCurrentPosition( async function(position){ try{ const gpsLat = position.coords.latitude; const gpsLng = position.coords.longitude; // ========================= // GEOCODE // ========================= const start = await geocode(from); const end = await geocode(to); // ========================= // REMOVE OLD LAYERS // ========================= [polyline,startMarker,endMarker,liveMarker] .forEach(function(layer){ if(layer) map.removeLayer(layer); }); // ========================= // START MARKER // ========================= startMarker = L.marker([start.lat,start.lng],{ icon:makeIcon('start') }).addTo(map); // ========================= // END MARKER // ========================= endMarker = L.marker([end.lat,end.lng],{ icon:makeIcon('end') }).addTo(map); // ========================= // GPS MARKER // ========================= liveMarker = L.circleMarker([gpsLat,gpsLng],{ radius:8, color:'#22c55e' }).addTo(map); // ========================= // ROUTE API // ========================= const response = await fetch( `https://api.openrouteservice.org/v2/directions/${selectedVehicle}/geojson`, { method:'POST', headers:{ 'Authorization':API_KEY, 'Content-Type':'application/json' }, body:JSON.stringify({ coordinates:[ [start.lng,start.lat], [end.lng,end.lat] ] }) } ); const data = await response.json(); // ========================= // DRAW ROUTE // ========================= polyline = L.geoJSON(data,{ style:{ color:'#3b82f6', weight:6 } }).addTo(map); map.fitBounds(polyline.getBounds()); // ========================= // SUMMARY // ========================= const summary = data.features[0].properties.summary; const km = (summary.distance / 1000).toFixed(2); const mins = Math.round(summary.duration / 60); // ========================= // COST // ========================= let rate = selectedVehicle === 'car' ? 20 : 3.5; const cost = (km * rate).toFixed(2); // ========================= // UI UPDATE // ========================= document.getElementById('lv-dist') .textContent = km; document.getElementById('lv-dur') .textContent = mins + 'm'; document.getElementById('f-dist') .textContent = km; document.getElementById('f-dur') .textContent = mins + ' min'; document.getElementById('f-name') .innerHTML = '👤 ' + name + ''; document.getElementById('f-start') .innerHTML = '🟢 From: ' + from; document.getElementById('f-end') .innerHTML = '🔴 To: ' + to; document.getElementById('result-card') .classList.add('show'); // ========================= // AUTO SAVE // ========================= tripData = { name:name, start:from, end:to, distance:km, duration:mins, cost:cost, gps_lat:gpsLat, gps_lng:gpsLng, vehicle:selectedVehicle }; await submitToZoho(); setStatus( 'done', 'Trip Saved', km + ' km completed' ); toast('Trip saved successfully'); } catch(err){ console.error(err); toast('Route calculation failed'); } }, function(error){ toast('GPS permission denied'); } ); }
